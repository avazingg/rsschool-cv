function int32ToIp(int32) {

    if (int32 == 0) {
        return '0.0.0.0'
    }
    else {
        let binary = int32.toString(2).match(/.{8}/g).join('.')
        let arr = binary.split('.')
        let arr1 = []
        for (let i = 0; i < arr.length; i++) {
            arr1.push(parseInt(arr[i], 2))
        }
        for (let j = 0; j < arr1.length; j++) {
            arr1[j] + '.'
        }
        return arr1.join('.')
    }
}
